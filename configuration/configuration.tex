\newcommand{\braket}[2]{\left\langle#1|#2\right\rangle}

\newcommand{\twovector}[2]{\begin{pmatrix} #1 \\ #2 \end{pmatrix}} % Vector de dim 2

% Funciones e info debajo de funciones
